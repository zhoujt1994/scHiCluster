def diff_ranksum(c1, c2):
    global data
    # print(c1, c2)
    fc = (data[c1].mean(axis=0) + 0.01) / (data[c2].mean(axis=0) + 0.01)
    pv = np.zeros(len(fc))
    for i in range(len(pv)):
        pv[i] = ranksums(data[c1][:, i], data[c2][:, i])[1]
    fdr = FDR(pv, 0.01, "fdr_bh")[1]
    return [c1, c2, fc, fdr]
